# Group the data by 'Segment' and aggregate Sales and Profit
sales_profit_by_segment = data.groupby('Segment').agg({'Sales': 'sum', 'Profit': 'sum'}).reset_index()

# Set the plot size
plt.figure(figsize=(12, 6))

# Create a bar plot for Sales and Profit
bar_width = 0.35
index = range(len(sales_profit_by_segment))

# Create bars for Sales with a bright red color
plt.bar(index, sales_profit_by_segment['Sales'], width=bar_width, label='Sales', color='red')

# Create bars for Profit with a bright yellow color, offset to the right
plt.bar([i + bar_width for i in index], sales_profit_by_segment['Profit'], width=bar_width, label='Profit', color='yellow')

# Customize the title and axis labels
plt.title('Sales and Profit Analysis by Customer Segment', fontsize=18, fontweight='bold')
plt.xlabel('Customer Segment', fontsize=12)
plt.ylabel('Amount', fontsize=12)
plt.xticks([i + bar_width / 2 for i in index], sales_profit_by_segment['Segment'])  # Center x-ticks

# Add a legend
plt.legend()

# Show the plot
plt.tight_layout()
plt.show()
